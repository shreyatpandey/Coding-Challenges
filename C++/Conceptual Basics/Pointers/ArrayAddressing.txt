Question-1:-
#include <stdio.h>
int main()
{
    int arr[5] = {1,2,3,4};
    printf("%p\n",arr);
    printf("%p\n",(arr+1));
    return 0;
}
// Output:- 
0x7fff44cb4240
0x7fff44cb4244
Address would always be = Original Address + sizeof(datatype)

Question-2:-
#include <stdio.h>
int main()
{
    int arr[4] = {1,2,3,4};
    printf("%x\n",*(arr)+1); //2 
    printf("%x\n",*(arr+1));//2
    printf("%x\n",*(arr+3));//4
    printf("%x\n",*(arr+4));//0

    return 0;
}
//Output
2
2
4
0
